39 template <class TPU> struct TPUBuild {
41  using type_elem = typename std::remove_reference_t<decltype((TPU{})[0])>;
42 
44  static constexpr size_t size_elem = sizeof(type_elem);

47  static constexpr size_t size = sizeof(TPU) / size_elem;
48 
50  using array = std::array<type_elem, size>;

52  template <class Fun, decltype(size)... Is>
53  static constexpr TPU make_helper(Fun f, std::index_sequence<Is...>) {
54  static_assert(std::is_invocable_v<Fun, type_elem>);
55  return TPU{f(Is)...};
56  }

59  inline constexpr TPU operator()(std::initializer_list<type_elem> il,
60  type_elem def) const {
61  HPCOMBI_ASSERT(il.size() <= size);
62  array res;
63  std::copy(il.begin(), il.end(), res.begin());
64  std::fill(res.begin() + il.size(), res.end(), def);
65  return reinterpret_cast<const TPU &>(res);
66  }
